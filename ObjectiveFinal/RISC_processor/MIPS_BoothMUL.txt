LD R1,0(R0)
LD R2,4(R0)
MOVE R3,R0
MOVE R4,R0
ADDI R5,R0,16

LOOP:
BZ R5,EXIT
ANDI R6,R2,1
BPL R6,LABEL_M
BZ R6,LABEL_P
BR NONE

LABEL_M:
BZ R4,MINUS 
BR NONE

LABEL_P:
BPL R4,PLUS
BR NONE

MINUS:
SUB R3,R3,R1
BR NONE

PLUS:
ADD R3,R3,R1

NONE:
MOVE R4,R6
ANDI R7,R3,1
SRLI R2,1
SLI R7,15
OR R2,R2,R7
SRAI R3,1
SUBI R5,1
BR LOOP

EXIT:
MOVE R1,R0
SLI R3,16
OR R1,R1,R3
ANDI R2,R2,-1
OR R1,R1,R2
MOVE R2,R1
HALT
